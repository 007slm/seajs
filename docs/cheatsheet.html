seajs.use 载入模块 

seajs.use('./a');

seajs.use('./a', function(a) {
  a.doSomething();
});

seajs.use(['./a', './b'], function(a, b) {
  a.doSomething();
  b.doSomething();
});

seajs.config 全局配置 

seajs.config({
  base: 'path/to/jslib/',
  alias: {
    'app': 'path/to/app/'
  },
  charset: 'utf-8',
  timeout: 20000,
  debug: false
});

seajs.alias alias 

seajs.alias('app', 'path/to/app');

define 模块书写格式 

define(function(require, exports, module) {
  // The module code goes here
});

require require 

define(function(require) {
  var $ = require('jquery');
  var a = require('./a');

  a.doSomething();
  // snip...
});

exports exports 

define(function(require, exports) {
  // snip...
  exports.name = 'a';
  exports.doSomething = function() {};
});

module module 

define(function(require, exports, module) {
  // snip...
  module.exports = {
    name: 'a'，
    doSomething: function() {};
  };
});

module.load module.load 

define(function(require, exports, module) {

  module.load('./b', function(b) {
    b.doSomething();
  });

  module.load(['./c', './d'], function(c, d) {
    // do something
  });

});
